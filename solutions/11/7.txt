% Parameters 
int: water_limit = 5000; % Liters 
int: land_limit = 400; % Square Meters 
int: manure_limit = 4000; % Kilograms 
int: seeds_limit = 50; % Kilograms 
int: water_per_veg_sqm = 10; % Liters 
float: seeds_per_veg_sqm = 0.1; % Kilograms 
int: yield_veg = 2; 
int: water_per_fruit_sqm = 15; % Liters 
float: seeds_per_fruit_sqm = 0.2; % Kilograms 
int: yield_fruit = 3; 
int: water_per_fruit2_sqm = 20; % Liters 
float: seeds_per_fruit2_sqm = 0.15; % Kilograms 
int: yield_fruit2 = 4; 
int: manure_per_sqm = 20; % Kilograms 
int: weight_veg_required = 200; % Kilograms 
int: weight_fruit_required = 150; % Kilograms 
int: weight_fruit2_required = 100; % Kilograms 
int: total_area_limit = 500; % Square Meters 

% Decision Variables 
var 0..total_area_limit: area_veg; 
var 0..total_area_limit: area_fruit; 
var 0..total_area_limit: area_fruit2; 

% Constraints 
constraint area_veg + area_fruit + area_fruit2 <= total_area_limit; 

constraint area_veg * water_per_veg_sqm + area_fruit * water_per_fruit_sqm + area_fruit2 * water_per_fruit2_sqm <= water_limit; 

constraint area_veg * manure_per_sqm + area_fruit * manure_per_sqm + area_fruit2 * manure_per_sqm <= manure_limit; 

constraint area_veg * seeds_per_veg_sqm + area_fruit * seeds_per_fruit_sqm + area_fruit2 * seeds_per_fruit2_sqm <= seeds_limit; 

constraint area_veg * yield_veg >= weight_veg_required; 
constraint area_fruit * yield_fruit >= weight_fruit_required; 
constraint area_fruit2 * yield_fruit2 >= weight_fruit2_required; 

% Objective (No optimization is needed) 
solve satisfy; 

% Output 
output [ "Area for Vegetables (sqm): " ++ show(area_veg) ++ "" ++ 
        "Area for Fruit (sqm): " ++ show(area_fruit) ++ "" ++ 
        "Area for Second Fruit (sqm): " ++ show(area_fruit2) ];
